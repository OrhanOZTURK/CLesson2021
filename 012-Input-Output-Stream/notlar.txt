#include <stdio.h>
#include <math.h>

int ndigit(int val)
{
    if(val == 0)
        return 1;

    int digit_count = 0;

    while (val) {
        ++digit_count;
        val /= 10;
    }

    return digit_count;
}

int main()
{
    int ival;
    printf("bir tam sayi giriniz : ");
    scanf("%d", &ival);

    printf("%d sayisi %d basamaklidir\n", ival, ndigit(ival));
    return 0;
}
